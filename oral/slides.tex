        % Colonne de gauche : Ton texte direct

                    \vspace{0.2cm}

                \end{block}

            \end{itemize}
        \end{column}

    \end{columns}

            \vspace{0.2cm}

                \item \textbf{Risque :} Perdre en mobilité.
                \item \textbf{IA :} Donne un poids \textbf{négligeable} au score (Poids 1).
            \end{itemize}
        \end{column}

        \begin{column}{0.45\textwidth}
            \textbf{En fin de partie :}
            \vspace{0.2cm}
            \begin{itemize}
                \item \textbf{Changement :} L'IA prend en compte le nombre de pions restant dès qu'il reste moins de 10 cases vides.
                \item \textbf{IA :} Le nombre de pions devient un critère \textbf{important} (Poids 100).
            \end{itemize}
        \end{column}
    \end{columns}
